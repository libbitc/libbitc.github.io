77  // Require minimal encodings for all push operations (OP_0... OP_16, OP_1NEGATE where possible, direct
78  // pushes up to 75 bytes, OP_PUSHDATA up to 255 bytes, OP_PUSHDATA2 for anything larger). Evaluating
79  // any other push causes the script to fail (BIP62 rule 3).
80  // In addition, whenever a stack element is interpreted as a number, it must be of minimal length (BIP62 rule 4).
81  // (softfork safe)
82  SCRIPT_VERIFY_MINIMALDATA = (1U << 6),
83 
84  // Discourage use of NOPs reserved for upgrades (NOP1-10)
85  //
86  // Provided so that nodes can avoid accepting or mining transactions
87  // containing executed NOP's whose meaning may change after a soft-fork,
88  // thus rendering the script invalid; with this flag set executing
89  // discouraged NOPs fails the script. This verification flag will never be
90  // a mandatory flag applied to scripts in a block. NOPs that are not
91  // executed, e.g. within an unexecuted IF ENDIF block, are *not* rejected.
92  SCRIPT_VERIFY_DISCOURAGE_UPGRADABLE_NOPS = (1U << 7),
93 
94  // Require that only a single stack element remains after evaluation. This changes the success criterion from
95  // "At least one stack element must remain, and when interpreted as a boolean, it must be true" to
96  // "Exactly one stack element must remain, and when interpreted as a boolean, it must be true".
97  // (softfork safe, BIP62 rule 6)
98  // Note: CLEANSTACK should never be used without P2SH or WITNESS.
